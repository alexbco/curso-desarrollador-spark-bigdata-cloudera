-- TABLA EMPLEADOS
-- ============================================================

empno INT PRIMARY KEY,
ename VARCHAR(20),
job VARCHAR(20),
mgr INT NULL,
hiredate DATE,
sal DECIMAL(10,2),
comm DECIMAL(10,2) NULL,
deptno INT NULL,
FOREIGN KEY (deptno) REFERENCES dept(deptno)
);
